      date_of_birth DATE

  id bigint [primary key]
  name varchar(255)
  phone varchar(20)
  address varchar(255)
      date_of_birth DATE
    expertise VARCHAR(255)

    id bigint [primary key]
    title VARCHAR(255)
    description TEXT
    level VARCHAR(50)
    duration int
    quantity int
    teacher_id bigint
}
